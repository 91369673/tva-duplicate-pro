# Befehle zum Hinzufügen des GitHub-Repositories

# Ersetzen Sie IHRUSERNAME durch Ihren GitHub-Benutzernamen
git remote add origin https://github.com/IHRUSERNAME/tva-duplicate-pro.git

# Oder wenn Sie SSH verwenden:
# git remote add origin git@github.com:IHRUSERNAME/tva-duplicate-pro.git

# Code hochladen
git push -u origin main

# Status überprüfen
git remote -v
